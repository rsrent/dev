import 'package:bms_flutter/src/components/primary_button.dart';
import 'package:bms_flutter/src/components/primary_button.dart';
import 'package:bms_flutter/src/widgets/loading_overlay.dart';
import 'package:bms_dart/models.dart';
import 'package:bms_flutter/translations.dart';
import 'package:dart_packages/streamer.dart';
import 'package:flutter/material.dart';
import 'package:flutter_bloc/flutter_bloc.dart';
import 'package:bms_flutter/src/widgets/absence_create_update_form.dart';
import 'package:bms_dart/absence_create_update_bloc.dart';
import 'package:rxdart/rxdart.dart';

class AbsenceCreateScreen extends StatelessWidget {
  final User user;
  final Streamer<bool> _loading = Streamer(seedValue: false);
  AbsenceCreateScreen({@required this.user});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(Translations.of(context).titleCreateAbsence),
      ),
      body: BlocProvider(
        builder: (context) {
          return AbsenceCreateUpdateBloc(user: user)..dispatch(PrepareCreate());
        },
        child: Builder(
          builder: (context) {
            var _bloc = BlocProvider.of<AbsenceCreateUpdateBloc>(context);
            return BlocListener(
              bloc: _bloc,
              listener: (context, AbsenceCreateUpdateState state) {
                _loading.update(state is Loading);

                if (state is CreateFailure) {
                  Scaffold.of(context)
                    ..removeCurrentSnackBar()
                    ..showSnackBar(SnackBar(
                      content:
                          Text(Translations.of(context).infoCreationFailed),
                    ));
                } else if (state is CreateSuccessful) {
                  Scaffold.of(context)
                    ..removeCurrentSnackBar()
                    ..showSnackBar(SnackBar(
                      content:
                          Text(Translations.of(context).infoCreationSuccessful),
                    ));
                  _bloc.dispatch(PrepareCreate());
                }
              },
              child: BlocBuilder(
                bloc: _bloc,
                builder: (context, AbsenceCreateUpdateState state) {
                  return bodyBuilder(context, state, _bloc);
                },
              ),
            );
          },
        ),
      ),
    );
  }

  Widget bodyBuilder(BuildContext context, AbsenceCreateUpdateState state,
      AbsenceCreateUpdateBloc bloc) {
    return Center(
      child: Stack(
        fit: StackFit.expand,
        children: <Widget>[
          SingleChildScrollView(
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.stretch,
              mainAxisAlignment: MainAxisAlignment.spaceEvenly,
              children: <Widget>[
                AbsenceCreateUpdateForm(),
                StreamBuilder<List<bool>>(
                  stream: Observable.combineLatest3(
                      bloc.canUserCreate.stream,
                      bloc.canUserRequest.stream,
                      bloc.formValid.stream,
                      (a, b, c) => [a, b, c]),
                  builder: (BuildContext context, AsyncSnapshot snapshot) {
                    var showCreate = snapshot.hasData && snapshot.data[0];
                    var showRequest = snapshot.hasData && snapshot.data[1];

                    var widgets = <Widget>[];

                    print(snapshot.data);

                    if (showCreate) {
                      widgets.add(PrimaryButton(
                        onPressed: () {
                          bloc.dispatch(CreateRequested(asRequest: false));
                        },
                        text: Translations.of(context).buttonCreate,
                        disabled: !snapshot.data[2],
                      ));
                    }
                    if (showRequest) {
                      widgets.add(PrimaryButton(
                        onPressed: () {
                          bloc.dispatch(CreateRequested(asRequest: true));
                        },
                        text: 'ANMOD',
                        disabled: !snapshot.data[2],
                      ));
                    }

                    return Column(
                      children: widgets,
                    );
                  },
                ),
              ],
            ),
          ),
          if (state is Loading) Center(child: CircularProgressIndicator())
        ],
      ),
    );
  }
}
